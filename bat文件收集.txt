@echo off
cd /d %~dp0
%1 start "" mshta vbscript:createobject("shell.application").shellexecute("""%~0""","::",,"runas",1)(window.close)&exit

命令大意就是，在指定文件夹jre1.8.0_201/bin/下运行java -jar referee-v1.0.jar命令。

还可以额外使用的命令，如：

指定运行窗口的标题：title 窗口标题名。
运行程序出错时防止程序自动退出窗口，一遍查看报错内容：在指令最后一行 添加 pause;
以上命令可以丰富为 ：

title referee
start jre1.8.0_201/bin/java -jar referee-v1.0.jar
pause;

批处理基础（九）接收用户输入信息   注：set /p命令中，/p表示允许将用户输入作为变量值，

@echo off
:again
cls
echo.
echo.
echo ######################请选择要执行的操作######################
echo ----------------------1、输入数字1并按回车，显示当前日期----------------------
echo ----------------------2、输入数字2并按回车，显示当前时间----------------------
echo ----------------------3、输入数字3并按回车，显示文件列表----------------------
echo ----------------------4、输入任何其它字符，终止批处理   ----------------------
echo.
echo.
echo 请选择要执行的操作
set /p num=

if "%num%"=="1" (
cls
echo.
echo.
echo 当前日期为：
date/t
pause
goto again
)

if "%num%"=="2" (
cls
echo.
echo.
echo 当前时间为：
time/t
pause
goto again
)

if "%num%"=="3" (
cls
echo.
echo.
echo 当前文件列表：
dir /w
pause
goto again
)

echo.
echo.
echo 输入了%num%字符，批处理自动推出。
